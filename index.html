// -- SETTING BATAS ALARM -- const BATAS_SUHU_PANAS = 35.0; // Contoh: Alarm bunyi jika > 35°C const BATAS_KELEMBAPAN_KERING = 20; // Contoh: Alarm bunyi jika < 20% // Fungsi untuk Bunyi Beep (Alarm) function playAlarmSound() { const audioCtx = new (window.AudioContext || window.webkitAudioContext)(); const oscillator = audioCtx.createOscillator(); const gainNode = audioCtx.createGain(); oscillator.connect(gainNode); gainNode.connect(audioCtx.destination); oscillator.type = 'sine'; oscillator.frequency.setValueAtTime(880, audioCtx.currentTime); // Nada tinggi gainNode.gain.setValueAtTime(0.1, audioCtx.currentTime); oscillator.start(); oscillator.stop(audioCtx.currentTime + 0.5); // Bunyi selama 0.5 detik } function fetchData() { if (!isMonitoring) return; db.ref("monitoring").once("value").then((snapshot) => { const data = snapshot.val(); if (data) { const s = parseFloat(data.suhu).toFixed(1); const k = data.kelembapan; const waktu = new Date().toLocaleTimeString(); // --- LOGIKA ALARM --- let statusSistem = "NORMAL"; let isAlarm = false; if (s > BATAS_SUHU_PANAS || k < BATAS_KELEMBAPAN_KERING) { statusSistem = "!! BAHAYA !!"; isAlarm = true; playAlarmSound(); // Bunyi Bip! } // Update Warna Background Stat jika Alarm const cardSuhu = document.querySelector('.bg-red'); const cardSoil = document.querySelector('.bg-blue'); if (isAlarm) { cardSuhu.style.backgroundColor = "#ff0000"; // Merah terang cardSuhu.style.boxShadow = "0 0 20px #ff0000"; } else { cardSuhu.style.backgroundColor = "#e74c3c"; // Merah normal cardSuhu.style.boxShadow = "none"; } // Update UI document.getElementById("txt-suhu").innerText = s; document.getElementById("txt-kelembapan").innerText = k; // Update Grafik if (myChart.data.labels.length > 15) { myChart.data.labels.shift(); myChart.data.datasets[0].data.shift(); myChart.data.datasets[1].data.shift(); } myChart.data.labels.push(waktu); myChart.data.datasets[0].data.push(s); myChart.data.datasets[1].data.push(k); myChart.update(); // Update Tabel const tableBody = document.getElementById("log-table-body"); const rowColor = isAlarm ? 'style="background-color: #ffeaea; font-weight: bold; color: red;"' : ''; const newRow = `${waktu}${s}${k}${statusSistem}`; tableBody.insertAdjacentHTML('afterbegin', newRow); } }); }
